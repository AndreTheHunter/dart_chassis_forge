Future<void> doc(
  IShell shell, {
  String output = "docs",
  String format = "html",
}) async {
  _whenDartProject('Doc', shell, () async {
    _log.info('Building Documentation...');
    shell.requireCommand('dartdoc');
    await shell.run('dartdoc --output $output --format $format');
  });
}
